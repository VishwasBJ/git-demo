“your_email@example.com” git config –list

git init git status git log

git add . git add filename

git commit -m “Initial commit” git commit -m “Updated code” git commit
–amend

remote -v git remote remove origin git remote set-url origin
https://github.com/username/repository.git

git push origin main git push -u origin main git push origin feature1
git push –force origin main

CLONING REPOSITORY

git branch git branch feature1 git checkout feature1 git checkout main

MERGE OPERATIONS

git merge feature1

DIFFERENCE BETWEEN BRANCHES

git revert HEAD

git pull origin main
